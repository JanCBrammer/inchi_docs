61/*************** Macros for retrieving requested INChI and INChI_Aux *****************************/
62/* S->pINChI[TAUT_YES] has info: */
63#define HAS_T(S) (S->pINChI[TAUT_YES] && S->pINChI[TAUT_YES]->nNumberOfAtoms)
64/* S->pINChI[TAUT_NON] has info: */
65#define HAS_N(S) (S->pINChI[TAUT_NON] && S->pINChI[TAUT_NON]->nNumberOfAtoms)
66
67/* S->pINChI[TAUT_YES] has tautomeric info: */
68#define HAS_TT(S) (S->pINChI[TAUT_YES] && S->pINChI[TAUT_YES]->nNumberOfAtoms && S->pINChI[TAUT_YES]->lenTautomer>0)
69/* S->pINChI[TAUT_YES] has non-taitomeric info: */
70#define HAS_TN(S) (S->pINChI[TAUT_YES] && S->pINChI[TAUT_YES]->nNumberOfAtoms && !S->pINChI[TAUT_YES]->lenTautomer)
71/* S->pINChI[TAUT_NON] has non-tautomeric info: */
72#define HAS_NN(S) (S->pINChI[TAUT_NON] && S->pINChI[TAUT_NON]->nNumberOfAtoms && !S->pINChI[TAUT_NON]->lenTautomer)
73#define GET_II(M,S) ((M==OUT_N1)? (HAS_TN(S)? TAUT_YES : HAS_NN(S)? TAUT_NON : -1): \
74 (M==OUT_T1 || M==OUT_TN)? (HAS_T(S) ? TAUT_YES : HAS_N(S) ? TAUT_NON : -1): \
75 (M==OUT_NN)? (HAS_NN(S)? TAUT_NON : HAS_TN(S)? TAUT_YES : -1): \
76 (M==OUT_NT)? ((HAS_TT(S) && HAS_NN(S)) ? TAUT_NON : -1) : -1)

82/* What is equal (ii = INChI_ITEM) */
83#define iiSTEREO 0x0001 /* stereo (sp2 or sp3) */
84#define iiSTEREO_INV 0x0002 /* inverted stereo (sp3) */
85#define iiNUMB 0x0004 /* numbering or inverted stereo numbering */
86#define iiEQU 0x0008 /* equivalence info */
87/* derived:
88 (iiSTEREO_INV | iiNUMB) = numbering of inverted stereo
89*/
90
91/* Additional info to what is equal (INCHI_ITEM_TYPE = iit) */
92#define iitISO 0x0010 /* Isotopic */
93#define iitNONTAUT 0x0020 /* Non-tautomeric */
94/* derived:
95 (iitISO | iitNONTAUT) = isotopic non-tautomeric
96*/
97
98/* Where is the equivalent item located (INChI_ITEM_EQUAL_TO = iiEq2) */
99#define iiEq2NONTAUT 0x0040 /* non-tautomeric */
100#define iiEq2ISO 0x0080 /* isotopic */
101#define iiEq2INV 0x0100 /* equal to inverted (stereo sp3) or to numbering of inverted stereo */
102
103#define iiEmpty 0x0200 /* item is empty while in the preceding layer the item is not empty */
