40/* InChI Extensible API Object Handles */
41/*
42 * These are the "handles" which can be used to refer to "Objects"
43 * used in the InChI Extensible API.
44 *
45 * To ensure that each handle has a different formal type, each is declared
46 * as a pointer to a C struct containing a dummy integer field (which is in
47 * fact not used). In practice, the values are simply cast from int, but
48 * this approach provides greater type security, and prevents handles for
49 * different objects getting mixed up.
50*/
51
52typedef struct { int dummy; } IXA_STATUS_HANDLE_STRUCT;
53typedef IXA_STATUS_HANDLE_STRUCT *IXA_STATUS_HANDLE;
54

56typedef IXA_MOL_HANDLE_STRUCT *IXA_MOL_HANDLE;
57
58typedef struct { int dummy; } IXA_INCHIBUILDER_HANDLE_STRUCT;
59typedef IXA_INCHIBUILDER_HANDLE_STRUCT *IXA_INCHIBUILDER_HANDLE;
60
61typedef struct { int dummy; } IXA_INCHIKEYBUILDER_HANDLE_STRUCT;
62typedef IXA_INCHIKEYBUILDER_HANDLE_STRUCT *IXA_INCHIKEYBUILDER_HANDLE;

65/* Types for Atom, Bond and Stereo Descriptor Identifiers */
66/*
67 * These types are for the identifiers for individual atoms, bonds and
68 * stereodescriptors in an IXA Molecule Object.
69 *
70 * To ensure that each identifier has a different formal type, each is
71 * declared as a pointer to a C struct containing a dummy integer field
72 * (which is in fact not used). In practice, the values are simply cast
73 * from int, but this approach provides greater type security, and
74 * prevents different sorts of identifier from getting mixed up.
75*/
76
