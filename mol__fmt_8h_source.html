231typedef struct A_MOL_FMT_ATOM
232{
233 double fx; /* F10.5; Generic */
234 double fy; /* F10.5; Generic */
235 double fz; /* F10.5; Generic */
236 char symbol[6]; /* aaa; Generic */
237
238 S_CHAR mass_difference; /* dd; (M_ISO) */
239 /* Generic: -3..+4 otherwise 0 or */
240 /* 127=most abund. isotope */
241 S_CHAR charge; /* ccc; (M CHG), */
242 /* Generic: 1=+3, 2=+2,3=+1, */
243 /* 4=doublet,5=-1,6=-2,7=-3 */
244 char radical; /* (M RAD) */
245 char stereo_parity; /* sss; Generic */
246#if ( MOL_FMT_QUERY == MOL_FMT_PRESENT )
247 char H_count_plus_1; /* hhh; Query; */
248 /* Hn means >= n H; */
249 /* H0 means no H */
250 char stereo_care; /* bbb; Query: 0=ignore; */
251 /* 1=must match */
252#endif
253 char valence; /* vvv: */
254 /* 0=no marking; (1..14)=(1..14); */
255 /* 15=zero valence.Number of bonds*/
256 /* includes bonds to impl. H's */
257
258#if ( MOL_FMT_CPSS == MOL_FMT_PRESENT )
259 char H0_designator; /* HHH: CPSS */
260 char reaction_component_type; /* rrr: */
261 /* CPSS: 1=reactant, */
262 /* 2=product, */
263 /* 3=intermediate */
264 char reaction_component_num; /* iii: CPSS: 0 to (n-1) */
265#endif
266#if ( MOL_FMT_REACT == MOL_FMT_PRESENT )
267 short atom_atom_mapping_num; /* mmm: Reaction: 1..255 */
268 char cInversionRetentionFlag; /* nnn: */
269 /* 1=inverted, 2=retained config.;*/
270 /* 0=property not applied */
271#endif
272#if ( MOL_FMT_REACT == MOL_FMT_PRESENT || MOL_FMT_QUERY == MOL_FMT_PRESENT )
273 char exact_change_flag; /* eee */
274#endif
275 char my_n_impH; /* number of implicit H calculated */
276 /* for adding H to strings in STDATA*/
277 char display_tom; /* Do not hide element's name */
278 /* (applies to C 7-25-98 DCh */
279 char atom_aliased_flag; /* Do not remove */
280 /* charge/radical/isotope if it */
281 /* is in the alias. 9-3-99 DCh */
282} MOL_FMT_ATOM;

285typedef struct A_MOL_FMT_BOND
286{
287 short atnum1; /* 111: First atom number: Generic */
288 short atnum2; /* 222: Second atom number:Generic */
289 char bond_type; /* ttt: */
290 /* 1,2,3=single, double, triple; */
291 /* 4=aromatic; */
292 /* 5=single or double; */
293 /* 6=single or aromatic; */
294 /* 7=double or aromatic; */
295 /* 8=any. */
296 /* Values 4-8 are for */
297 /* SSS queries only */
298 char bond_stereo; /* sss: */
299 /* Single bonds: */
300 /* 0=not stereo, 1=up, */
301 /* 4=either, 6=down */
302 /* Double bonds: */
303 /* 0=use x,y,z to */
304 /* determine cis/trans, */
305 /* 3=cis or trans (either) */
306 /* xxx: not used */
307
308#if ( MOL_FMT_QUERY == MOL_FMT_PRESENT )
309 char bond_topology; /* rrr: */
310 /* 0=either, 1=ring, 2=chain: */
311 /* SSS queries only */
312#endif
313#if ( MOL_FMT_REACT == MOL_FMT_PRESENT )
314 char react_center_status; /* ccc: */
315 /* 0 = unmarked, */
316 /* 1 = a center, */
317 /* -1 = not a center; */
318 /* Additional: */
319 /* 2 = no charge, */
320 /* 4 = bond made/broken, */
321 /* 8 = bond order changes */
322 /* 12=4+8; 5=4+1, 9=8+1, 13=12+1 */
323 /* 12=4+8; 5=4+1, 9=8+1, 13=12+1 */
324 /* are also possible */
325#endif
326} MOL_FMT_BOND;
