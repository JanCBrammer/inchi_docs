39typedef enum tagInchiCompareDiffBits {
40 INCHIDIFF_ZERO = 0x00000000,
41 INCHIDIFF_PROBLEM = 0x00000001, /* severe: at least one InChI does not exist */
42 INCHIDIFF_NUM_AT = 0x00000001, /* severe: different number of atoms in the skeleton */
43 INCHIDIFF_ATOMS = 0x00000001, /* severe: diiferent types of skeleton atoms */
44 INCHIDIFF_NUM_EL = 0x00000001, /* severe: formulas differ in another element */
45 INCHIDIFF_CON_LEN = 0x00000001, /* severe: different connection table lengths */
46 INCHIDIFF_CON_TBL = 0x00000001, /* severe: different connection tables */
47 INCHIDIFF_POSITION_H = 0x00000002, /* difference in non-taut (Mobile-H) or all H (Fixed-H) location/number */
48 INCHIDIFF_MORE_FH = 0x00000004, /* extra fixed H */
49 INCHIDIFF_LESS_FH = 0x00000004, /* missing fixed H */
50 INCHIDIFF_MORE_H = 0x00000008, /* formulas differ in number of H */
51 INCHIDIFF_LESS_H = 0x00000008, /* formulas differ in number of H */
52 INCHIDIFF_NO_TAUT = 0x00000010, /* restored structure has no taut groups while the original InChI has some */
53 INCHIDIFF_WRONG_TAUT = 0x00000020, /* restored has tautomerism while the original does not have it */
54 INCHIDIFF_SINGLE_TG = 0x00000040, /* restored has 1 taut. group while the original InChI has multiple tg */
55 INCHIDIFF_MULTIPLE_TG = 0x00000080, /* restored has multiple tg while the original InChI has only one tg */
56 INCHIDIFF_EXTRA_TG_ENDP = 0x00000100, /* extra tautomeric endpoint(s) in restored structure */
57 INCHIDIFF_MISS_TG_ENDP = 0x00000100, /* one or more tg endpoint is not in the restored structure */
58 INCHIDIFF_DIFF_TG_ENDP = 0x00000100, /* lists of tg endpoints are different */
59 INCHIDIFF_NUM_TG = 0x00000200, /* different number of tautomeric groups */
60 INCHIDIFF_TG = 0x00000200, /* different tautomeric groups */
61 INCHIDIFF_NUM_ISO_AT = 0x00000400, /* ?severe: restored struct. has different number of isotopic atoms */
62 INCHIDIFF_ISO_AT = 0x00000400, /* ?severe: restored struct. has different locations/isotopes of isotopic atoms */
63 INCHIDIFF_REM_ISO_H = 0x00000800, /* isotopic H removed */
64 INCHIDIFF_MOB_ISO_H = 0x00001000, /* different number of mobile exchangeable isotopic H */
65 INCHIDIFF_CHARGE = 0x00002000, /* restored structure has different charge */
66 INCHIDIFF_REM_PROT = 0x00004000, /* proton(s) removed/added from the restored structure */
67 INCHIDIFF_MOBH_PROTONS = 0x00008000, /* different proton balance */
68 INCHIDIFF_SC_INV = 0x00010000, /* restores structure has different inversion stereocenter mark */
69 INCHIDIFF_SC_PARITY = 0x00020000, /* restored structure has stereoatoms or allenes with different parity */
70 INCHIDIFF_SC_EXTRA_UNDF = 0x00040000, /* restored structure has extra undefined stereocenter(s) */
71 INCHIDIFF_SC_EXTRA = 0x00080000, /* restored structure has extra stereocenter(s) */
72 INCHIDIFF_SC_MISS_UNDF = 0x00100000, /* restored structure has not some undefined stereocenter(s) */
73 INCHIDIFF_SC_MISS = 0x00200000, /* restored structure has not some stereocenters that are not undefined */
74 INCHIDIFF_SB_PARITY = 0x00400000, /* restored structure has stereobonds or cumulenes with different parity */
75 INCHIDIFF_SB_EXTRA_UNDF = 0x00800000, /* restored structure has extra undefined stereobond(s) */
76 INCHIDIFF_SB_EXTRA = 0x01000000, /* restored structure has extra stereobond(s) */
77 INCHIDIFF_SB_MISS_UNDF = 0x02000000, /* restored structure has not some undefined stereocenters */
78 INCHIDIFF_SB_MISS = 0x04000000, /* restored structure has not some stereobonds that are not undefined */
79 INCHIDIFF_COMP_HLAYER = 0x08000000, /* Restored component has Mobile-H layer instead of both Mobile-H & Fixed-H or both instead of one */
80 INCHIDIFF_COMP_NUMBER = 0x10000000, /* wrong number of components */
81 INCHIDIFF_STR2INCHI_ERR = 0x20000000 /* Restored structure to InChI conversion error */
82 /* reserved
83 0x40000000
84 0x80000000
85 */
86} INCHIDIFF;
