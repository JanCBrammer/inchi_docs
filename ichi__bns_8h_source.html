387/*********************************************************************************
388 bChangeFlow:
389 1 => change flow inside the BNS search
390 3 => change flow inside the BNS search and undo the flow change in the BNS structure here
391 4 => change bonds in the structure according to the flow
392 8 => make altern. bonds in the structure
393
394 Note: (bChangeFlow & 1) == 1 is needed for multiple runs
395**********************************************************************************/
396
397/* "EF" = "Edge Flow" */
398#define BNS_EF_CHNG_FLOW 1 /* change Balanced Network (BN) flow inside the BNS search */
399#define BNS_EF_RSTR_FLOW 2 /* undo BN flow changes after BNS */
400#define BNS_EF_CHNG_RSTR (BNS_EF_CHNG_FLOW | BNS_EF_RSTR_FLOW)
401#define BNS_EF_CHNG_BONDS 4 /* change bonds in the structure according to the BN flow */
402#define BNS_EF_ALTR_BONDS 8 /* make altern. bonds in the structure if the flow has changed */
403#define BNS_EF_UPD_RAD_ORI 16 /* update BN flow0 & Atom radical values:
404 flow0 := flow, radical:=st_cap - st_flow */
405#define BNS_EF_SET_NOSTEREO 32 /* in combination with BNS_EF_ALTR_BONDS only:
406 ALT12 bond cannot be stereogenic */
407#define BNS_EF_UPD_H_CHARGE 64 /* update charges and H-counts according to change flow to c- and t-group vertices */
408
409#define BNS_EF_SAVE_ALL (BNS_EF_CHNG_FLOW | BNS_EF_CHNG_BONDS | BNS_EF_UPD_RAD_ORI)
410#define BNS_EF_ALTR_NS (BNS_EF_ALTR_BONDS | BNS_EF_SET_NOSTEREO)
