102typedef struct tagInputAtom
103{
104 char elname[ATOM_EL_LEN]; /* name of chemical element */
105 U_CHAR el_number; /* number of the element in the Periodic Table */
106 AT_NUMB neighbor[MAXVAL]; /* positions (from 0) of the neighbors in the inp_ATOM array*/
107 AT_NUMB orig_at_number; /* original atom number */
108 AT_NUMB orig_compt_at_numb; /* original atom number within the component */
109 /* before terminal H removal */
110 S_CHAR bond_stereo[MAXVAL]; /* 1=Up,4=Either,6=Down; this atom is at the pointing wedge,*/
111 /* negative => on the opposite side; 3=Either double bond */
112 U_CHAR bond_type[MAXVAL]; /* 1..4; 4="aromatic", should be discouraged on input */
113 S_CHAR valence; /* for most chemists, it is coordination number, CN; */
114 /* number of bonds = number of neighbors */
115 S_CHAR chem_bonds_valence; /* for most chemists, it is what usually called valence; */
116 /* sum of bond types (type 4 needs special treatment) */
117 S_CHAR num_H; /* number of implicit hydrogens, including D and T */
118 S_CHAR num_iso_H[NUM_H_ISOTOPES]; /* number of implicit 1H, 2H(D), 3H(T) < 16 */
119 S_CHAR iso_atw_diff; /* =0 => natural isotopic abundances */
120 /* >0 => (mass) - (mass of the most abundant isotope) + 1 */
121 /* <0 => (mass) - (mass of the most abundant isotope) */
122 S_CHAR charge; /* charge */
123 S_CHAR radical; /* RADICAL_SINGLET, RADICAL_DOUBLET, or RADICAL_TRIPLET */
124 S_CHAR bAmbiguousStereo;
125 S_CHAR cFlags; /* AT_FLAG_ISO_H_POINT */
126 AT_NUMB at_type; /* ATT_NONE, ATT_ACIDIC */
127 AT_NUMB component; /* number of the structure component > 0 */
128 AT_NUMB endpoint; /* id of a tautomeric group */
129 AT_NUMB c_point; /* id of a positive charge group */
130 double x;
131 double y;
132 double z;
133 /* 0D parities (originally were used with CML) */
134 S_CHAR bUsed0DParity; /* bit=1 => stereobond; bit=2 => stereocenter */
135 /* 0D tetrahedral parity */
136 S_CHAR p_parity;
137 AT_NUMB p_orig_at_num[MAX_NUM_STEREO_ATOM_NEIGH];
138 /* 0D bond parities */
139 S_CHAR sb_ord[MAX_NUM_STEREO_BONDS]; /* stereo bond/neighbor ordering number, starts from 0 */
140 /* neighbors on both sides of stereobond have same sign=> trans/T/E, diff. signs => cis/C/Z */
141 S_CHAR sn_ord[MAX_NUM_STEREO_BONDS]; /* ord. num. of the neighbor adjacent to the SB; */
142 /* starts from 0; */
143 /* -1 means removed explicit H */
144 /* neighbors on both sides of stereobond have same parity => trans/T/E/2, diff. parities => cis/C/Z/1 */
145 S_CHAR sb_parity[MAX_NUM_STEREO_BONDS];
146 AT_NUMB sn_orig_at_num[MAX_NUM_STEREO_BONDS]; /* orig. at number of sn_ord[] neighbors */
147
148#if ( FIND_RING_SYSTEMS == 1 )
149 S_CHAR bCutVertex;
150 AT_NUMB nRingSystem;
151 AT_NUMB nNumAtInRingSystem;
152 AT_NUMB nBlockSystem;
153
154#if ( FIND_RINS_SYSTEMS_DISTANCES == 1 )
155 AT_NUMB nDistanceFromTerminal; /* terminal atom or ring system has 1, next has 2, etc. */
156#endif
157
158#endif
159} inp_ATOM;
